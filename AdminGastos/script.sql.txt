DROP TABLE IF EXISTS ADMGASTOS.GASTOS;
  CREATE TABLE ADMGASTOS.GASTOS (
    ID INT(11) NOT NULL,
    FECHA TIMESTAMP NOT NULL, 
    CANTIDAD INTEGER NOT NULL,
    DESC_PROD VARCHAR(255) NOT NULL , 
    TOTAL DECIMAL(11,4) NOT NULL,
    ESTATUS INT(2)    NOT NULL , 
    FECHA_ALTA TIMESTAMP   NOT NULL , 
    FECHA_ULT_ACT TIMESTAMP  DEFAULT LOCALTIMESTAMP  NOT NULL , 
    PRIMARY KEY(ID)
  ) ENGINE=INNODB;
